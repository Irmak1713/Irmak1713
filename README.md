  <div class="buttons">
    <button onclick="feedCat('fish')">🐟 Fish</button>

    <button onclick="feedCat('chicken')">🍗 Chicken</button>

  <script>
    const cat = document.getElementById("cat");
    const message = document.getElementById("message");

    function feedCat(food) {
      cat.classList.add("talk");
      setTimeout(() => cat.classList.remove("talk"), 500);

      switch(food) {
        case 'fish':
          message.textContent = "Yum yum! I love fish! 🐟😻";
          break;
        case 'candy':
          message.textContent = "Eww... I don’t like candy! 🍬😾";
          break;
        case 'chicken':
          message.textContent = "Delicious chicken! 🍗😽";
          break;
        default:
          message.textContent = "What is this?! 😾";
      }
    }
  </script>
